
CREATE TABLE public.perfil (
                ID VARCHAR(50)NOT NULL,
                nome VARCHAR(50) NOT NULL,
                email VARCHAR(50) NOT NULL,
                CONSTRAINT perfil_pk PRIMARY KEY (ID)
);

CREATE SEQUENCE public.itens_cod_item_seq;

CREATE TABLE public.itens (
                cod_item NUMERIC(6) NOT NULL DEFAULT nextval('public.itens_cod_item_seq'),
                ID VARCHAR(50) NOT NULL,
                nome_eq VARCHAR(50) NOT NULL,
                gps VARCHAR(50) NOT NULL,
                validade VARCHAR(50) NOT NULL,
                email_op VARCHAR(50),
                CONSTRAINT itens_pk PRIMARY KEY (cod_item)
);


ALTER SEQUENCE public.itens_cod_item_seq OWNED BY public.itens.cod_item;

ALTER TABLE public.itens ADD CONSTRAINT perfil_itens_fk
FOREIGN KEY (ID)
REFERENCES public.perfil (ID)
ON DELETE NO ACTION
ON UPDATE NO ACTION
NOT DEFERRABLE;
